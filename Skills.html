from pybricks.hubs import PrimeHub
from pybricks.pupdevices import Motor, ColorSensor
from pybricks.parameters import Port, Direction
from pybricks.robotics import DriveBase
from pybricks.tools import wait
# Initialize hub
hub = PrimeHub()
# Motors (with direction)
left_motor = Motor(Port.B, Direction.COUNTERCLOCKWISE) # Left Wheel
right_motor = Motor(Port.E, Direction.CLOCKWISE) # Right Wheel
drive_base = DriveBase(left_motor, right_motor, wheel_diameter=56, axle_track=112)
# Sensor
color_sensor = ColorSensor(Port.C) # Ground Scanner
# Threshold for line (adjust based on surface)
threshold = 50 # Reflection % for line edge
while True:
    reflection = color_sensor.reflection()
    error = reflection - threshold
    turn_rate = error * 2 # Proportional control
    drive_base.drive(200, turn_rate)
    if color_sensor.color() == Color.RED:
        drive_base.stop()
        break
    wait(10)
